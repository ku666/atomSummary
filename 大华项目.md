### 3. commonJs和AMD
	commonJs: require()引入是同步的，等引入后才能使用
	var math = require("math.js");
	math.add(a,b); // 加载时间过长会造成假死状态

	AMD: 使用异步加载 require.js curl.js
	require(["math.js"],function(math){
		math.add(a,b);
	})
